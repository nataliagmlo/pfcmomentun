def calculate_users_list influences
	users_list = []
	top = 0
	pos = 0
	while top < 10 and pos < influences.size
		if not user_in(top, influences[pos].user, users_list)
			users_list[top] = influences[pos]
			top += 1
		end
		pos += 1
	end
	users_list
end

def show
	@user = nil
	if params["search"] != ""
		@user = User.name_like params["search"]
		@user = @user.first
	    if @user!=nil
	      redirect_to user_path(@user)
	    else
	    	calculate_list()
	    	@error = "El usuario '" + params["search"] + "' no existe"
	      	render :action => :index
	    end
	else
		calculate_list()
	    @error = "Debes introducir un nombre de usuario"
	    render :action => :index
	end
end

def user_in top, user, users
	i = 0
	while i < top
		if users[i].user.equals(user)
			return true
		end
		i += 1
	end 
	return false
end
