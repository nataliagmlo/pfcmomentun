def to_s
  s = "User:"
  s += "\n\tuser_id: " + user_id.to_s
  s += "\n\tname: " + name.to_s
  s += "\n\treal_name: " + real_name.to_s 
  s += "\n\tsubscribers: " + subscribers.to_s
  s += "\n\tsubscriptions: " + subscriptions.to_s
  s += "\n\tlast_mention_at: " + last_mention_at.to_s if last_mention_at!=nil
  s
end
