def to_s
  s = "Influence:"
  s += "\n\tuser_id: " + user_id.to_s
  s += "\n\tdate: " + date.to_s
  s += "\n\tacceleration: " + acceleration.to_s 
  s += "\n\tvelocity: " + velocity.to_s
  s += "\n\taudience: " + audience.to_s 
  s += "\n\tcreated_at: " + created_at.to_s
  s
end
